# read error correction tool
# refer to fig3 for overview of Racon consensus process https://pmc.ncbi.nlm.nih.gov/articles/PMC5411768/
# Racon uses Partial Order Alignment (POA) to identify the most likely true sequence, replacing erroneous bases and building consensus sequence
PATH=/work/soghigian_lab/apps/racon/build/bin:$PATH
racon -t 8 \
  sample42_q18_trimmed.fastq.gz \
  all_to_reps.paf \
  sample42.haplotypes.reps.fasta \
  > haplotypes.racon.fasta
  
conda activate /work/soghigian_lab/apps/conda/envs/medaka_2.2.2

# medaka is a deep learning polisher
# header of fastq file will indicate which basecalling model used
# zless sample.fastq
# basecall_model_version_id=dna_r10.4.1_e8.2_400bps_sup@v5.2.0 rc

medaka_consensus -t 8 \
  -i sample42_q18_trimmed.fastq.gz \
  -d haplotypes.racon.fasta \
  -o medaka_vsearch_out \
  -m r1041_e82_400bps_sup_v5.2.0

cp ./medaka_vsearch_out/consensus.fasta haplotypes.polished.fasta

#bsub.py 9 hap_map -q long "bash map.sh"
{
  echo -e "Run\tHaplotype\tReads"
  P="sample42.paf"
  RUN=$(basename "$P" .paf)
  awk 'BEGIN{FS="\t"; OFS="\t"}
       {key=$6"\t"$1; if(!(key in seen)) { seen[key]=1; c[$6]++ }}
       END{for(h in c) print "'"$RUN"'", h, c[h]}' "$P"
} > run_haplotype_counts.tsv
